USVideos <- read.csv("data/USvideos.csv")

#Data Cleaning
yt_data <- USvideos

yt_data$video_id <- NULL
yt_data$tags <- NULL
yt_data$trending_date <- NULL
yt_data$thumbnail_link <- NULL
yt_data$description <- NULL
yt_data$video_error_or_removed <- NULL
yt_data$category_id <- NULL

yt_data <- na.omit(yt_data)

ct_data <- yt_data[
  !duplicated(
    yt_data[, c("title", "publish_time", "channel_title")]
  ),
]

yt_data1 <- ct_data$ratings_disabled == "False"
ct_data1 <- ct_data[yt_data1, ]

ct_data2 <- ct_data1$comments_disabled == "False"
bt_data <- ct_data1[ct_data2, ]

rownames(bt_data) <- NULL

bt_data$comments_disabled <- NULL
bt_data$ratings_disabled <- NULL

bt_data$dislikes[bt_data$dislikes == 0] <- 1
bt_data$comment_count[bt_data$comment_count == 0] <- 1
bt_data$likes[bt_data$likes == 0] <- 1

dislike_data <- bt_data |>
  dplyr::select(
    likes,
    dislikes,
    comment_count,
    views
  )

#Create Linear Model
lm_dislike_log <- lm(
  log(dislikes) ~
    log(likes) +
    log(views) +
    log(comment_count),
  data = dislike_data
)

Box-Cox Transformation

\[ \text{dislikes} = \beta_0 + \beta_1 \log(\text{likes}) + \beta_2 \log(\text{views}) + \beta3 \log(\text{comment\_count}) +\varepsilon_i \]

#Create Full Model

Final Model

So our final model after collecting the betas for our model is:

\[ \log(\text{dislikes}) = -4.06283 - 0.18455\log(\text{likes}) + 0.55405\log(\text{views}) + 0.63115\log(\text{comment\_count}) + \varepsilon_i \] Can also be written as:

\[ \text{dislikes} = e^{-4.06283} \cdot \frac{ \text{views}^{0.55405} (\text{comment\_count}^{0.63115}) }{ \text{likes}^{0.18455} } \cdot e^{\varepsilon_i} \]

output$pred_table <- renderTable({

nd <- data.frame(
  likes = input$likes,
  comment_count = input$comments,
  views = input$views
)

  lm_dislike_log,
  newdata = nd,
  interval = "prediction",
  level = 0.95
)
